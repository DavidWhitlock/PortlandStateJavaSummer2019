016   *                                                                                
017   * @param name                                                                    
018   *        The student's name                                                      
019   * @param classes                                                                 
020   *        The names of the classes the student is taking.  A student              
021   *        may take zero or more classes.                                          
022   * @param gpa                                                                     
023   *        The student's grade point average                                       
024   * @param gender                                                                  
025   *        The student's gender ("male", "female", or "other", case insensitive)
026   */                                                                               
027  public Student(String name, ArrayList<String> classes, double gpa, Gender gender) {
028    super(name);
029
030    if (name == null) {
031      throw new NullPointerException("Name cannot be null");
032    }
033
034    if (gpa < 0.0) {
035      throw new GPAOutOfBoundsException("GPA cannot be less than zero");
036
037    } else if (gpa > 4.0) {
038      throw new GPAOutOfBoundsException("GPA cannot be more than 4.0");
039    }
040
041    this.gender = gender;
042  }

053   * Returns a <code>String</code> that describes this

057    throw new UnsupportedOperationException("Not implemented yet");
058  }

060  /**
061   * Main program that parses the command line, creates a
062   * <code>Student</code>, and prints a description of the student to
063   * standard out by invoking its <code>toString</code> method.
064   */
065  public static void main(String[] args) {
066    if (args.length == 0) {
067      System.err.println("Missing command line arguments");
068      System.exit(1);
069    }
070
071    String gpaString = args[2];
072    try {
073      double gpa = Double.parseDouble(gpaString);
074
075    } catch (NumberFormatException ex) {
076      System.err.println("Invalid gpa: " + gpaString);
077    }
078
079    String genderString = args[1];
080    Gender gender = genderFromString(genderString);
081    if (gender == null) {
082      System.err.println("Unsupported gender");
083      System.exit(1);
084    }
085
086
087    System.exit(0);
088  }
089
090  private static Gender genderFromString(String genderString) {
091    switch (genderString.toLowerCase()) {
092      case "female":
093        return Gender.FEMALE;
094      case "male":
095        return Gender.MALE;
096      case "other":
097        return Gender.OTHER;
098      default:
099        return null;
100    }
101  }
102
103  public Gender getGender() {
