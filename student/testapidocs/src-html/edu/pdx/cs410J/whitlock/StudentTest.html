001package edu.pdx.cs410J.whitlock;
002
003import org.junit.Test;
004
005import java.util.ArrayList;
006
007import static org.hamcrest.MatcherAssert.assertThat;
008import static org.hamcrest.core.IsEqual.equalTo;
009
010/**
011 * Unit tests for the Student class.  In addition to the JUnit annotations,
012 * they also make use of the <a href="http://hamcrest.org/JavaHamcrest/">hamcrest</a>
013 * matchers for more readable assertion statements.
014 */
015public class StudentTest
016{
017
018  private Student createStudentNamed(String name) {
019    return new Student(name, new ArrayList<>(), 0.0, Gender.OTHER);
020  }
021
022  @Test
023  public void studentNamedPatIsNamedPat() {
024    String name = "Pat";
025    var pat = createStudentNamed(name);
026    assertThat(pat.getName(), equalTo(name));
027  }
028
029  @Test(expected = NullPointerException.class)
030  public void whenNameIsNullThrowANullPointerException() {
031    String name = null;
032    createStudentNamed(name);
033  }
034
035
036  @Test(expected = GPAOutOfBoundsException.class)
037  public void whenGPAIsLessThanZeroThrowGPAOutOfBoundsException() {
038    double gpa = -1.0;
039    new Student("name", new ArrayList<>(), gpa, Gender.OTHER);
040  }
041
042  @Test(expected = GPAOutOfBoundsException.class)
043  public void whenGPAIsGreaterThanFourThrowGPAOutOfBoundsException() {
044    double gpa = 4.1;
045    new Student("name", new ArrayList<>(), gpa, Gender.OTHER);
046  }
047
048  @Test
049  public void whenGenderIsFemaleStudentIsFemale() {
050    Student student = new Student("name", new ArrayList<>(), 1.0, Gender.FEMALE);
051    assertThat(student.getGender(), equalTo(Gender.FEMALE));
052  }
053
054  @Test
055  public void whenGenderIsMaleStudentIsMale() {
056    Student student = new Student("name", new ArrayList<>(), 1.0, Gender.MALE);
057    assertThat(student.getGender(), equalTo(Gender.MALE));
058  }
059
060  @Test
061  public void whenGenderIsOtherStudentIsOther() {
062    Student student = new Student("name", new ArrayList<>(), 1.0, Gender.OTHER);
063    assertThat(student.getGender(), equalTo(Gender.OTHER));
064  }
065
066
067}
